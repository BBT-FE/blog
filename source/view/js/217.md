```js
//  此法实现多重继承，是利用call/apply实现多重继承，核心就是用Function类的call/apply方法去绑定新的类，使新的类实例化后的对象继承了该属性及方法
var s1 = function(name){
  this.name = name;
}

var Student = function(name, score){
  s1.call(this, name);
  this.score = score;
}
Student.prototype.constructor = Student;
var s = new Student('jack',  '100');
alert(s.name);

```
